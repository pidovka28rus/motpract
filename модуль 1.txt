-- Таблица типов материалов
CREATE TABLE MATERIAL_TYPE (
    MATERIAL_TYPE_ID INTEGER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    MATERIAL_TYPE_NAME VARCHAR(100) UNIQUE
);

-- Таблица материалов
CREATE TABLE MATERIAL (
    MATERIAL_ID INTEGER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    MATERIAL_NAME VARCHAR(255) NOT NULL,
    MATERIAL_TYPE_ID INTEGER NOT NULL,
    UNIT_PRICE DECIMAL(15,2),
    QUANTITY_IN_STOCK DECIMAL(15,2),
    MIN_QUANTITY DECIMAL(15,2),
    PACKAGING_QUANTITY DECIMAL(15,2),
    UNIT_OF_MEASUREMENT VARCHAR(20),
    FOREIGN KEY (MATERIAL_TYPE_ID) REFERENCES MATERIAL_TYPE(MATERIAL_TYPE_ID)
);

-- Таблица типов продукции
CREATE TABLE PRODUCT_TYPE (
    PRODUCT_TYPE_ID INTEGER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    PRODUCT_TYPE_NAME VARCHAR(100) NOT NULL UNIQUE,
    COEFFICIENT DECIMAL(10,2)
);

-- Таблица продукции
CREATE TABLE PRODUCT (
    PRODUCT_ARTICLE VARCHAR(20) PRIMARY KEY,
    PRODUCT_NAME VARCHAR(255) NOT NULL,
    MIN_COST_FOR_PARTNER DECIMAL(15,2),
    PRODUCT_TYPE_ID INTEGER,
    FOREIGN KEY (PRODUCT_TYPE_ID) REFERENCES PRODUCT_TYPE(PRODUCT_TYPE_ID)
);

-- Таблица связи "материалы — продукция"
CREATE TABLE MATERIAL_PRODUCT (
    MATERIAL_ID INTEGER,
    PRODUCT_ARTICLE VARCHAR(20),
    REQUIRED_QUANTITY DECIMAL(15,2),
    PRIMARY KEY (MATERIAL_ID, PRODUCT_ARTICLE),
    FOREIGN KEY (MATERIAL_ID) REFERENCES MATERIAL(MATERIAL_ID),
    FOREIGN KEY (PRODUCT_ARTICLE) REFERENCES PRODUCT(PRODUCT_ARTICLE)
);
